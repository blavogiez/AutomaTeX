change completement l'editeur pour integrer l'editeur ace. Tu suis l'exemple suivant : import webview

# HTML contenant Ace Editor
html_ace_editor = """
<!DOCTYPE html>
<html>
<head>
    <title>Ace Editor</title>
    <style>
        html, body, #editor {
            height: 100%;
            margin: 0;
            padding: 0;
        }
    </style>
</head>
<body>
    <div id="editor"># Code Python ici\nprint("Hello from Ace!")</div>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/ace/1.23.4/ace.js"></script>
    <script>
        var editor = ace.edit("editor");
        editor.setTheme("ace/theme/monokai");
        editor.session.setMode("ace/mode/python");
        editor.setOptions({ fontSize: "14pt", wrap: true });

        window.getEditorContent = function() {
            return editor.getValue();
        };
    </script>
</body>
</html>
"""

# Création de la fenêtre webview avec contenu HTML embarqué
webview.create_window("Ace Editor dans pywebview", html=html_ace_editor, width=800, height=600)
webview.start(). Mais tu vas tout adapter à l'appli pour que le nouvel éditeur soit ace et que tout soit adapté et fonctionne comme avant. Vas y !!
